```python
import math
from typing import List


def calc_probabilty(prob: List[int]):
    """–§—É–Ω–∫—Ü–∏—è —Ä–∞—Å—á—ë—Ç–∞ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏ –≤–æ–∑–Ω–∏–∫–Ω–æ–≤–µ–Ω–∏—è –æ—à–∏–±–∫–∏ –Ω–∞ —Å–µ—Ä–≤–µ—Ä–µ"""
    result = [0.0] * len(prob)
    amount = sum(prob)
    for i in range(len(prob)):
        result[i] = prob[i] / amount
    return result


if __name__ == '__main__': # —Ç–æ—á–∫–∞ –∑–∞–ø—É—Å–∫–∞ –≤–∞—à–µ–≥–æ —Ä–µ—à–µ–Ω–∏—è
    servers = int(input()) # –≤–≤–æ–¥ –¥–∞–Ω–Ω—ã—Ö –æ –∫–æ–ª–∏—á–µ—Å—Ç–≤–µ —Å–µ—Ä–≤–µ—Ä–æ–≤
    probabilities = [
        math.prod(map(int, input().split())) for _ in range(servers)
    ] # —Ä–∞—Å—á—ë—Ç –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω—ã—Ö –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–µ–π –¥–ª—è –∫–∞–∂–¥–æ–≥–æ —Å–µ—Ä–≤–µ—Ä–∞
    result = calc_probabilty(probabilities) # –≤—ã–∑–æ–≤ —Ñ—É–Ω–∫—Ü–∏–∏ –¥–ª—è —Ä–∞—Å—á—ë—Ç–∞ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏ –æ—à–∏–±–∫–∏
    for probability in result:
        print(probability) # –≤—ã–≤–æ–¥ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞
```
